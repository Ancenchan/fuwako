// 1. 修改初始化部分 // 这里的逻辑是：如果 Kuroshiro 直接是构造函数就用它，如果不是就找它的 .default const KuroshiroClass = window.Kuroshiro.default || window.Kuroshiro; const kuroshiro = new KuroshiroClass(); // 同理，Analyzer 也可能存在这个问题 const AnalyzerClass = window.KuromojiAnalyzer.default || window.KuromojiAnalyzer; const analyzer = new AnalyzerClass({ dictPath: "https://cdn.jsdelivr.net/npm/kuromoji@0.1.2/dict" }); // 2. 确保在库完全加载后再执行初始化 async function init() { try { // 检查库是否真的加载到了 window 对象上 if (!window.Kuroshiro || !window.KuromojiAnalyzer) { console.error("库文件加载失败，请检查网络"); return; } await kuroshiro.init(analyzer); document.getElementById('loadingOverlay').classList.add('hidden'); console.log("Kuroshiro 初始化成功！"); } catch (err) { console.error("初始化详细错误:", err); alert("词典加载失败，请尝试刷新页面。错误详情已打印在控制台。"); } } init();
